349 constexpr auto parse_nanos(std::string_view sv) noexcept
350 {
351  // clang-format off
352  constexpr int c[] = {
353  0,
354  100'000'000,
355  10'000'000,
356  1'000'000,
357  100'000,
358  10'000,
359  1'000,
360  100,
361  10,
362  1
363  };
364  // clang-format on
365 
366  // Truncate to ensure that we process no more than 9 decimal places.
367  sv = sv.substr(0, 9);
368  auto it = sv.begin(), end = sv.end();
369 
370  int ns{0};
371  if (isdigit(*it)) {
372  ns = *it++ - '0';
373  while (it != end && isdigit(*it)) {
374  ns *= 10;
375  ns += *it++ - '0';
376  }
377  }
378  return Nanos{ns * c[it - sv.begin()]};
379 }
380 static_assert(parse_nanos("000000001") == 1ns);

388 constexpr std::optional<Nanos> parse_time_only(std::string_view sv) noexcept
389 {
390  using namespace std::chrono;
391 
392  // clang-format off
393  if (sv.size() < 8
394  || !isdigit(sv[0])
395  || !isdigit(sv[1])
396  || sv[2] != ':'
397  || !isdigit(sv[3])
398  || !isdigit(sv[4])
399  || sv[5] != ':'
400  || !isdigit(sv[6])
401  || !isdigit(sv[7])) {
402  // Invalid format.
403  return {};
404  }
405  // clang-format on
406  const hours h{(sv[0] - '0') * 10 + sv[1] - '0'};
407  const minutes m{(sv[3] - '0') * 10 + sv[4] - '0'};
408  const seconds s{(sv[6] - '0') * 10 + sv[7] - '0'};
409  Nanos ns{h + m + s};
410  if (sv.size() > 8) {
411  if (sv[8] != '.') {
412  // Invalid delimiter.
413  return {};
414  }
415  ns += parse_nanos(sv.substr(9));
416  }
417  return ns;
418 }
419 static_assert(*parse_time_only("12:00:00"sv) == 12h);
