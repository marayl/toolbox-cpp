27 void run_reactor(Reactor& r, ThreadConfig config, const std::atomic<bool>& stop)
28 {
29  sig_block_all();
30  pthread_setname_np(pthread_self(), config.name.c_str());
31  TOOLBOX_NOTICE << "started " << config.name << " thread";
32  try {
33  while (!stop.load(std::memory_order_acquire)) {
