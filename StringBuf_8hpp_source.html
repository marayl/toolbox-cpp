98  void append(const char* rdata, std::size_t rlen) noexcept
99  {
100  rlen = std::min(rlen, MaxN - len_);
101  if (rlen > 0) {
102  std::memcpy(buf_ + len_, rdata, rlen);
103  len_ += rlen;
104  }
105  }
106  int compare(const char* rdata, std::size_t rlen) const noexcept
107  {
108  int result{std::memcmp(buf_, rdata, std::min(size(), rlen))};
109  if (result == 0) {
110  result = toolbox::compare(size(), rlen);
111  }
112  return result;
113  }
